import sys
import os
import re
import datetime
if (sys.argv[1] == '-ping'): 
    result=os.popen("ping -c 1 " + sys.argv[2]).read()
    result=re.findall(r"time=(.*) ms", result)
    print(result[0])
elif (sys.argv[1] == '-simple_print'): 
    print(sys.argv[2])
elif (sys.argv[1] == '1'): 
    print('Barankov Anton Vladislavovich')
elif (sys.argv[1] == '2'): 
    print(datetime.date.today())
else: 
    print(f"unknown input: {sys.argv[1]}")
